\section{Problem}
ABC and ADC are two right triangles with common hypotenuse AC. Prove that
$\angle CAD$ = $\angle CBD$.
\section{Solution}
Consider $\triangle ABC$ such that
\begin{align}
\vec{A} = \myvec{-a\\0} \\
\vec{C} = \myvec{a \\ 0} \\
\end{align}
Both the triangles ($\triangle{ABC} $ and $\triangle{ADC}$) are right angled.\\
Therefore, we can say that $\vec{B}$ and $\vec{D}$ lie on circle having $\vec{A}$ and $\vec{C}$ as its diametric end-points.\\
Equation of this circle is given by:
\begin{align}
\norm{\vec{x}-\brak{\frac{\vec{A}+\vec{C}}{2}}}^2 &= \norm{ \frac{\vec{A} -\vec{C}}{2}}^2 \\
\norm{\vec{x}}^2 = a^2
\end{align}
Points on this circle can be represented as
\begin{align}
\vec{x} = a\myvec{\cos{\theta}\\\sin{\theta}}
\end{align}
Let
\begin{align}
\vec{B} &= a\myvec{\cos{\theta_1}\\\sin{\theta_1}}\\
\vec{D} &= a\myvec{\cos{\theta_2}\\\sin{\theta_2}}
\end{align}
\begin{align}
\cos{\angle{CAD}} &= \frac{ \brak{\vec{C}-\vec{A}}^{\top} \brak{ \vec{D} -\vec{A}} }{\norm{ \vec{C} - \vec{A}} \norm{\vec{D} - \vec{A}}} \\ &= \sqrt{1+ \cos{\theta_1}}
\end{align}
\begin{align}
\cos{\angle{CBD}} &= \frac{ \brak{\vec{C}-\vec{B}}^{\top} \brak{ \vec{D} -\vec{B}} }{\norm{ \vec{C} - \vec{B}} \norm{\vec{D} - \vec{B}}}\\ &= \sqrt{1+ \cos{\theta_1}}
\end{align}
\begin{align}
\implies\;\; \angle{CAD} = \angle{CBD}
\end{align}
