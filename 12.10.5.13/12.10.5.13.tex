\section{PROBLEM 1}
1.  The scalar product of the vector $\myvec{1\\1\\1}$ with a unit vector along the sum of vectors  $\myvec{2\\4\\-5}$ and  $\myvec{\lambda\\2\\3}$  is equal to one. Find the value of $\lambda$.
\section{SOLUTION:}
Let,
\begin{align}
\vec{a}=\myvec{1\\1\\1}\ ; \vec{b}=\myvec{2\\4\\-5}\ ;\vec{c}=\myvec{\lambda\\2\\3}\\
\vec{b}+\vec{c} &= \myvec{2\\4\\-5} + \myvec{\lambda\\2\\3}\\ &= \myvec{2+\lambda\\6\\-2} \\
\text{and}, \norm{\vec{b} + \vec{c}} &= \sqrt{(2+\lambda)^2 + 6^2 + 2^2}\\
&= \sqrt{(2^2+2\times2\times\lambda+\lambda^2) + 36 + 4} \\
&= \sqrt{ \lambda^2 + 4\lambda + 44} 
\end{align}
Here, unit vector along $\vec{b} + \vec{c}$ is given by,
\begin{align}
\vec{v} &= \frac{\vec{b} + \vec{c}}{\norm{\vec{b} + \vec{c}}}\\
&=\frac{\myvec{2+\lambda\\6\\-2}}{ \sqrt{ \lambda^2 + 4\lambda + 44} }
\end{align}
It is given that,
\begin{align}
\vec{a}.\vec{v} = 1\\
\implies \myvec{1\\1\\1}\myvec{\frac{2+\lambda}{\sqrt{ \lambda^2 + 4\lambda + 44}}\\ \\\frac{6}{\sqrt{ \lambda^2 + 4\lambda + 44}}\\ \\\frac{-2}{\sqrt{ \lambda^2 + 4\lambda + 44}}} = 1\\
\implies \frac{2+\lambda+6-2}{\sqrt{ \lambda^2 + 4\lambda + 44}} = 1\\
\implies 2+\lambda+6-2= \sqrt{ \lambda^2 + 4\lambda + 44}\\
\implies \lambda +6 = \sqrt{ \lambda^2 + 4\lambda + 44}
\end{align}
squaring both sides, we get,
\begin{align}
(\lambda +6)^2 = \lambda^2 + 4\lambda + 44\\
\lambda^2 +12\lambda +36 = \lambda^2 + 4\lambda + 44\\
8\lambda = 8\\
\lambda = 1
\end{align}
